通用修饰器
对每个类型都有一个修饰器形式，怎么记得下来？所以就有了这个”万能修饰器”:

def w_test(func):
    def inner(*args, **kwargs):
        ret = func(*args, **kwargs)
        return ret

    return inner

@w_test
def test():
print(‘test called’)

@w_test
def test1():
print(‘test1 called’)
return ‘python’

@w_test
def test2(a):
print(‘test2 called and value is %d ‘ % a)

test()
test1()
test2(9)

输出:

#test called
#test1 called
#test2 called and value is 9
